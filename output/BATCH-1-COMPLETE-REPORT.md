"""
修复后的用户管理系统 - 安全版本
"""
import os
import json
import bcrypt
import secrets
from typing import Optional, List, Dict
from datetime import datetime
from contextlib import contextmanager
import sqlite3

class UserManager:
    """线程安全的用户管理类"""
    
    def __init__(self, db_path: str):
        self.db_path = db_path
        self._init_db()
    
    def _init_db(self):
        """初始化数据库表"""
        with self._get_conn() as conn:
            conn.execute('''
                CREATE TABLE IF NOT EXISTS users (
                    id INTEGER PRIMARY KEY AUTOINCREMENT,
                    name TEXT UNIQUE NOT NULL,
                    password_hash TEXT NOT NULL,
                    email TEXT UNIQUE NOT NULL,
                    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
                )
            ''')
    
    @contextmanager
    def _get_conn(self):
        """上下文管理器确保连接正确关闭"""
        conn = sqlite3.connect(self.db_path)
        try:
            yield conn
            conn.commit()
        finally:
            conn.close()
    
    def add_user(self, name: str, password: str, email: str) -> Dict:
        """添加用户 - 使用bcrypt哈希"""
        password_hash = bcrypt.hashpw(
            password.encode(), 
            bcrypt.gensalt(rounds=12)
        ).decode()
        
        with self._get_conn() as conn:
            cursor = conn.execute(
                "INSERT INTO users (name, password_hash, email) VALUES (?, ?, ?)",
                (name, password_hash, email)
            )
            return {
                'id': cursor.lastrowid,
                'name': name,
                'email': email
            }
    
    def login(self, name: str, password: str) -> bool:
        """验证登录 - 时序安全比较"""
        with self._get_conn() as conn:
            row = conn.execute(
                "SELECT password_hash FROM users WHERE name = ?",
                (name,)
            ).fetchone()
            
            if not row:
                # 执行虚假哈希防止时序攻击
                bcrypt.hashpw(b'dummy', bcrypt.gensalt())
                return False
            
            return bcrypt.checkpw(password.encode(), row[0].encode())
