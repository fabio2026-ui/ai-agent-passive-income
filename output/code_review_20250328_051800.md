## 7️⃣ 代码示例: 重构后的内容工厂

```python
# content_factory/refactored_factory.py

from abc import ABC, abstractmethod
from dataclasses import dataclass
from pathlib import Path
from typing import List, Protocol
import asyncio
import aiofiles


@dataclass(frozen=True)
class ContentConfig:
    """内容生成配置"""
    output_dir: Path
    batch_size: int
    platform: str


class ContentGenerator(Protocol):
    """内容生成器接口"""
    async def generate(self, topic: dict, index: int) -> tuple[str, str]:
        """返回 (文件名, 内容)"""
        ...


class BaseContentGenerator(ABC):
    """内容生成器基类"""
    
    def __init__(self, config: ContentConfig):
        self.config = config
    
    async def generate_batch(self, topics: List[dict]) -> None:
        """批量生成内容"""
        tasks = [
            self._generate_and_save(topic, i)
            for i, topic in enumerate(topics)
        ]
        await asyncio.gather(*tasks, return_exceptions=True)
    
    async def _generate_and_save(self, topic: dict, index: int) -> None:
        """生成并保存单个内容"""
        filename, content = await self.generate(topic, index)
        filepath = self.config.output_dir / filename
        await self._write_file(filepath, content)
    
    async def _write_file(self, filepath: Path, content: str) -> None:
        """异步写入文件"""
        filepath.parent.mkdir(parents=True, exist_ok=True)
        async with aiofiles.open(filepath, 'w', encoding='utf-8') as f:
            await f.write(content)
    
    @abstractmethod
    async def generate(self, topic: dict, index: int) -> tuple[str, str]:
        pass
```
