## 2. 安全问题

### 2.1 不安全的密码哈希 ⚠️ 严重

```python
# 问题代码
hash = hashlib.md5(pwd.encode()).hexdigest()
```

**风险**:
- MD5已被破解，不安全
- 没有使用盐值(salt)，易受彩虹表攻击
- 哈希速度快，易受暴力破解

**建议**: 使用`bcrypt`、`argon2`或`scrypt`。

### 2.2 SQL注入漏洞 ⚠️ 严重

```python
# database_module.py:26
def query(self, sql, params=()):
    full_sql = sql % params if params else sql
    cursor.execute(full_sql)

# database_module.py:37
def get_user_by_name(self, username):
    query = "SELECT * FROM users WHERE name='" + username + "'"
```

**风险**: 攻击者可以注入恶意SQL，窃取或删除数据。

### 2.3 使用eval() ⚠️ 严重

```python
# database_module.py:63
def parse_config(self, config_str):
    return eval(config_str)
```

**风险**: 允许执行任意代码，严重的远程代码执行漏洞。

### 2.4 硬编码密钥 ⚠️ 严重

```python

"""用户管理模块 - 优化版本.

提供用户创建、认证和管理功能，遵循安全最佳实践。
"""

import json
import os
import secrets
import hashlib
from datetime import datetime
from pathlib import Path
from typing import Optional, Protocol
import bcrypt


class UserRepository(Protocol):
    """用户存储接口."""
    
    def save(self, user: dict) -> None: ...
    def find_by_name(self, name: str) -> Optional[dict]: ...
    def find_by_id(self, user_id: int) -> Optional[dict]: ...
    def delete(self, user_id: int) -> bool: ...
    def get_all(self) -> list[dict]: ...


class JsonUserRepository:
    """JSON文件实现的用户存储."""
    
    def __init__(self, db_path: Path) -> None:
        self._db_path = db_path
        self._users: list[dict] = []
        self._load()
    
    def _load(self) -> None:
        """从文件加载用户数据."""
        if self._db_path.exists():
            with open(self._db_path, 'r', encoding='utf-8') as f:
                self._users = json.load(f)
    
    def _save(self) -> None:
        """保存用户数据到文件."""
        with open(self._db_path, 'w', encoding='utf-8') as f:
            json.dump(self._users, f, indent=2, ensure_ascii=False)
    
    def save(self, user: dict) -> None:
        """保存用户."""
        self._users.append(user)
        self._save()
    
    def find_by_name(self, name: str) -> Optional[dict]:
        """根据用户名查找用户."""
        return next((u for u in self._users if u['name'] == name), None)
    
    def find_by_id(self, user_id: int) -> Optional[dict]:
        """根据ID查找用户."""
        return next((u for u in self._users if u['id'] == user_id), None)
    
    def delete(self, user_id: int) -> bool:
        """删除用户."""
        for i, user in enumerate(self._users):
            if user['id'] == user_id:
                del self._users[i]
                self._save()
                return True
        return False
    
    def get_all(self) -> list[dict]:
        """获取所有用户."""
        return self._users.copy()


class UserManager:
    """用户管理器."""
    
    def __init__(self, repository: UserRepository) -> None:
        self._repo = repository
        self._user_index: dict[str, dict] = {}
        self._rebuild_index()
    
    def _rebuild_index(self) -> None:
        """重建用户索引以加速查找."""
        self._user_index = {
            u['name']: u for u in self._repo.get_all()
        }
    
    def _hash_password(self, password: str) -> str:
        """使用bcrypt哈希密码."""
        salt = bcrypt.gensalt(rounds=12)
        return bcrypt.hashpw(password.encode('utf-8'), salt).decode('utf-8')
    
    def _verify_password(self, password: str, hashed: str) -> bool:
        """验证密码."""
        return bcrypt.checkpw(
            password.encode('utf-8'),
            hashed.encode('utf-8')
        )
    
    def add_user(self, name: str, password: str, email: str) -> dict:
        """添加新用户.
        
        Args:
            name: 用户名
            password: 明文密码
            email: 邮箱地址
            
        Returns:
            创建的用户信息
            
        Raises:
            ValueError: 用户名已存在或输入无效
        """
        if not name or not password or not email:
            raise ValueError("用户名、密码和邮箱不能为空")
        
        if '@' not in email:
            raise ValueError("邮箱格式无效")
        
        if self.find_user(name):
            raise ValueError(f"用户 '{name}' 已存在")
        
        user = {
            'id': secrets.token_hex(8),
            'name': name,
            'password_hash': self._hash_password(password),
            'email': email,
            'created_at': datetime.utcnow().isoformat(),
            'updated_at': datetime.utcnow().isoformat()
        }
        
        self._repo.save(user)
        self._user_index[name] = user
        
        # 返回时排除密码哈希
        return {k: v for k, v in user.items() if k != 'password_hash'}
    
    def find_user(self, name: str) -> Optional[dict]:
        """查找用户."""
        return self._user_index.get(name)
    
    def authenticate(self, name: str, password: str) -> bool:
        """验证用户凭据."""
        user = self._user_index.get(name)
        if not user:
            # 固定时间防止用户枚举攻击
            bcrypt.checkpw(b'dummy', bcrypt.gensalt())
            return False
        return self._verify_password(password, user['password_hash'])
    
    def delete_user(self, user_id: int) -> bool:
        """删除用户."""
        result = self._repo.delete(user_id)
        if result:
            self._rebuild_index()
        return result
    
    def get_all_users(self) -> list[dict]:
        """获取所有用户（不包含密码）."""
        users = self._repo.get_all()
        return [
            {k: v for k, v in u.items() if k != 'password_hash'}
            for u in users
        ]


class UserExporter:
    """用户数据导出器."""
    
    @staticmethod
    def export_to_csv(users: list[dict], filepath: Path) -> None:
        """导出用户到CSV文件."""
        import csv
        
        with open(filepath, 'w', newline='', encoding='utf-8') as f:
            if users:
                writer = csv.DictWriter(f, fieldnames=users[0].keys())
                writer.writeheader()
                writer.writerows(users)
```

```python
"""数据库操作模块 - 优化版本."""

import json
import sqlite3
from contextlib import contextmanager
from pathlib import Path
from typing import Any, Optional
from dataclasses import dataclass


@dataclass
class DatabaseConfig:
    """数据库配置."""
    db_path: Path
    timeout: float = 30.0


class DatabaseError(Exception):
    """数据库操作错误."""
    pass


class Database:
    """线程安全的数据库连接管理."""
    
    def __init__(self, config: DatabaseConfig) -> None:
        self._config = config
        self._local = threading.local()
    
    @contextmanager
    def connection(self):
        """获取数据库连接的上下文管理器."""
        if not hasattr(self._local, 'conn'):
            self._local.conn = sqlite3.connect(
                self._config.db_path,
                timeout=self._config.timeout
            )
            self._local.conn.row_factory = sqlite3.Row
        
        try:
            yield self._local.conn
        except sqlite3.Error as e:
            self._local.conn.rollback()
            raise DatabaseError(f"数据库错误: {e}") from e
        finally:
            if hasattr(self._local, 'conn'):
                self._local.conn.close()
                delattr(self._local, 'conn')
    
    def execute(
        self,
        sql: str,
        parameters: tuple = ()
    ) -> sqlite3.Cursor:
        """执行参数化SQL语句."""
        with self.connection() as conn:
            cursor = conn.execute(sql, parameters)
            conn.commit()
            return cursor
    
    def fetchone(
        self,
        sql: str,
        parameters: tuple = ()
    ) -> Optional[sqlite3.Row]:
        """查询单条记录."""
        with self.connection() as conn:
            cursor = conn.execute(sql, parameters)
            return cursor.fetchone()
    
    def fetchall(
        self,
        sql: str,
        parameters: tuple = ()
    ) -> list[sqlite3.Row]:
        """查询多条记录."""
        with self.connection() as conn:
            cursor = conn.execute(sql, parameters)
            return cursor.fetchall()


class UserRepository:
    """用户数据访问对象."""
    
    def __init__(self, db: Database) -> None:
        self._db = db
        self._ensure_table()
    
    def _ensure_table(self) -> None:
        """确保用户表存在."""
        self._db.execute("""
            CREATE TABLE IF NOT EXISTS users (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                name TEXT UNIQUE NOT NULL,
                password_hash TEXT NOT NULL,
                email TEXT UNIQUE NOT NULL,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
    
    def get_by_name(self, name: str) -> Optional[dict]:
        """根据用户名获取用户 - 使用参数化查询防止SQL注入."""
        row = self._db.fetchone(
            "SELECT * FROM users WHERE name = ?",
            (name,)
        )
        return dict(row) if row else None


class DataProcessor:
    """优化后的数据处理类."""
    
    def __init__(self, max_cache_size: int = 1000) -> None:
        from functools import lru_cache
        self._cache = lru_cache(maxsize=max_cache_size)(self._expensive_operation)
    
    def process_items(
        self,
        items: list[dict[str, Any]]
    ) -> list[dict[str, Any]]:
        """处理项目列表."""
        processed = []
        for item in items:
            # 使用get()避免KeyError
            name = item.get('name')
            value = item.get('value', 0)
            
            if name is None:
                continue
            
            # 缓存计算结果
            result = self._cache(value)
            processed.append({'name': name, 'result': result})
        
        return processed
    
    def _expensive_operation(self, x: float) -> float:
        """耗时操作 - 结果被LRU缓存."""
        import time
        time.sleep(0.01)  # 模拟耗时操作
        return x * 2
    
    def safe_divide(self, x: float, y: float) -> float:
        """安全除法 - 防止除零错误."""
        if y == 0:
            raise ValueError("除数不能为零")
        return x / y
    
    def parse_config(self, config_str: str) -> dict:
        """安全解析JSON配置 - 替代eval."""
        try:
            return json.loads(config_str)
        except json.JSONDecodeError as e:
            raise ValueError(f"配置解析错误: {e}")
```

```python
"""API处理模块 - 优化版本."""

import http.server
import json
import logging
import secrets
import urllib.parse
from dataclasses import dataclass
from http.cookies import SimpleCookie
from typing import Callable, Optional

# 配置日志
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


@dataclass
class SessionConfig:
    """Session配置."""
    cookie_name: str = "session_id"
    max_age: int = 3600  # 1小时
    http_only: bool = True
    secure: bool = True
    same_site: str = "Strict"


class SessionManager:
    """安全的Session管理."""
    
    def __init__(self, config: SessionConfig) -> None:
        self._config = config
        self._sessions: dict[str, dict] = {}
    
    def create_session(self, user_id: str) -> str:
        """创建新session."""
        session_id = secrets.token_urlsafe(32)
        self._sessions[session_id] = {
            'user_id': user_id,
            'created_at': time.time()
        }
        return session_id
    
    def get_session(self, session_id: str) -> Optional[dict]:
        """获取session."""
        session = self._sessions.get(session_id)
        if session:
            # 检查过期
            if time.time() - session['created_at'] > self._config.max_age:
                del self._sessions[session_id]
                return None
        return session
    
    def build_cookie(self, session_id: str) -> str:
        """构建安全的Set-Cookie头."""
        cookie = SimpleCookie()
        cookie[self._config.cookie_name] = session_id
        cookie[self._config.cookie_name]['HttpOnly'] = self._config.http_only
        cookie[self._config.cookie_name]['Secure'] = self._config.secure
        cookie[self._config.cookie_name]['SameSite'] = self._config.same_site
        cookie[self._config.cookie_name]['Max-Age'] = self._config.max_age
        cookie[self._config.cookie_name]['Path'] = '/'
        return cookie.output(header='')


class APIHandler(http.server.BaseHTTPRequestHandler):
    """安全的API处理器."""
    
    routes: dict[str, Callable] = {}
    
    @classmethod
    def register(cls, path: str, handler: Callable) -> None:
        """注册路由处理器."""
        cls.routes[path] = handler
    
    def _send_json(self, data: dict, status: int = 200) -> None:
        """发送JSON响应."""
        self.send_response(status)
        self.send_header('Content-Type', 'application/json')
        self.end_headers()
        self.wfile.write(json.dumps(data).encode('utf-8'))
    
    def _send_error(self, message: str, status: int = 400) -> None:
        """发送错误响应."""
        self._send_json({'error': message}, status)
    
    def _parse_body(self) -> dict:
        """安全解析请求体."""
        content_length = self.headers.get('Content-Length')
        if not content_length:
            return {}
        
        try:
            length = int(content_length)
            if length > 1024 * 1024:  # 限制1MB
                raise ValueError("请求体过大")
            
            body = self.rfile.read(length)
            return json.loads(body.decode('utf-8'))
        except (ValueError, json.JSONDecodeError) as e:
            logger.warning(f"解析请求体失败: {e}")
            return {}
    
    def do_GET(self) -> None:
        """处理GET请求."""
        parsed = urllib.parse.urlparse(self.path)
        path = parsed.path
        query = urllib.parse.parse_qs(parsed.query)
        
        handler = self.routes.get(path)
        if not handler:
            self._send_error('Not Found', 404)
            return
        
        try:
            result = handler(query)
            self._send_json(result)
        except Exception as e:
            logger.exception("处理请求失败")
            self._send_error('Internal Server Error', 500)
    
    def log_message(self, format: str, *args) -> None:
        """安全日志记录 - 过滤敏感信息."""
        message = format % args
        # 过滤敏感信息
        filtered = self._filter_sensitive(message)
        logger.info(filtered)
    
    def _filter_sensitive(self, message: str) -> str:
        """过滤敏感信息."""
        import re
        # 过滤密码等敏感字段
        patterns = [
            (r'password[=:]\S+', 'password=***'),
            (r'token[=:]\S+', 'token=***'),
        ]
        for pattern, replacement in patterns:
            message = re.sub(pattern, replacement, message, flags=re.I)
        return message


class OrderService:
    """优化的订单服务."""
    
    def __init__(self, inventory_service, pricing_service) -> None:
        self._inventory = inventory_service
        self._pricing = pricing_service
        self._orders: dict[int, dict] = {}
        self._lock = threading.Lock()
        self._id_counter = 0
    
    def create_order(
        self,
        user_id: str,
        items: list[dict]
    ) -> dict:
        """创建订单 - 带库存检查和并发控制."""
        with self._lock:
            # 验证库存
            for item in items:
                if not self._inventory.check_stock(
                    item['product_id'],
                    item['quantity']
                ):
                    raise ValueError(f"库存不足: {item['product_id']}")
            
            # 计算价格
            total = self._pricing.calculate_total(items)
            
            self._id_counter += 1
            order = {
                'id': self._id_counter,
                'user_id': user_id,
                'items': items,
                'total': total,
                'status': 'pending',
                'created_at': time.time()
            }
            
            # 扣减库存
            for item in items:
                self._inventory.decrease_stock(
                    item['product_id'],
                    item['quantity']
                )
            
            self._orders[order['id']] = order
            return order
    
    def cancel_order(self, order_id: int, user_id: str) -> bool:
        """取消订单 - 带权限检查."""
        with self._lock:
            order = self._orders.get(order_id)
            if not order:
                return False
            
            # 权限检查
            if order['user_id'] != user_id:
                raise PermissionError("无权取消此订单")
            
            if order['status'] != 'pending':
                raise ValueError("订单状态不允许取消")
            
            # 恢复库存
            for item in order['items']:
                self._inventory.increase_stock(
                    item['product_id'],
                    item['quantity']
                )
            
            order['status'] = 'cancelled'
            return True
```
