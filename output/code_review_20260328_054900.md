```python

# 时间复杂度: O(n²) - 双重循环
# 空间复杂度: O(n)
def process_data(data_list):
    result=[]
    for i in range(len(data_list)):      # O(n)
        for j in range(len(data_list)):  # O(n) - 嵌套循环!
            if i!=j:
                if data_list[i]==data_list[j]:  # 重复比较
                    result.append(data_list[i])
    return result
```

**问题分析**:
- 该函数存在严重的算法复杂度问题
- 可以使用 `collections.Counter` 或 `set` 优化到 O(n)

**优化建议**:
```python
from collections import Counter

def process_data_optimized(data_list):
    """优化后的数据处理 - O(n) 复杂度"""
    counter = Counter(data_list)
    return [item for item, count in counter.items() if count > 1]
```

    def __init__(self):

    def get_from_cache(self, key):

**修复方案**:
```python
from functools import lru_cache
import time

class DataProcessor:
    def __init__(self, max_cache_size: int = 1000, ttl: int = 3600):
        self.cache = {}
        self.cache_time = {}
        self.max_cache_size = max_cache_size
        self.ttl = ttl
    
    def get_from_cache(self, key):
        # 检查过期
        if key in self.cache:
            if time.time() - self.cache_time.get(key, 0) < self.ttl:
                return self.cache[key]
            else:
                del self.cache[key]
                del self.cache_time[key]
        
        # LRU 清理
        if len(self.cache) >= self.max_cache_size:
            oldest = min(self.cache_time, key=self.cache_time.get)
            del self.cache[oldest]
            del self.cache_time[oldest]
        
        value = self._expensive_operation(key)
        self.cache[key] = value
        self.cache_time[key] = time.time()
        return value
```
